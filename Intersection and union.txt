#include <iostream.h>
struct node
   {
    int data;
    node* link;
   };
node *L1head , *L2head,*L1last,*L2last,*cur1,*cur2,*L3head ,*L3last,*cur3;
 int s=0;


void create ()
{
L1head = new node ;
cur1=L1head ;
for(int i=1;i<=1;i++)
    {
     cur1->link=new node ;
     cur1=cur1->link;
    }
L1last=cur1;
L1last->link=NULL;



L2head = new node ;
cur2=L2head ;
for(int i=1;i<=1;i++)
    {
     cur2->link=new node ;
     cur2=cur2->link;
    }
L2last=cur2;
L2last->link=NULL;



}
