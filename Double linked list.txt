#include <iostream.h>
struct node
{
int data ;
node *before , *after ;
};
typedef node *ptr;
ptr head , last , cura, curb;

void create_D()
{
head = new node ();
head->before=NULL;
cura=head;
curb=head;
for(int i=1;i<=4;i++)
    {
     cura->after=new node();
     curb=cura;
     cura=cura->after;
     cura->before=curb;
    }
last = cura;
last->after=NULL;
}

void dellast_D()
{
 last=last->before;

 last->after=NULL;
}
